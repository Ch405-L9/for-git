// lib/metadata.ts
import type { Metadata } from 'next'

export interface MetadataConfig {
  title: string
  description: string
  canonical?: string
  ogImage?: string
  keywords?: string[]
  local?: {
    city: string
    region: string
    postalCode?: string
    latitude?: string
    longitude?: string
  }
  service?: string
  noindex?: boolean
}

export function generatePageMetadata(config: MetadataConfig): Metadata {
  const {
    title,
    description,
    canonical,
    ogImage = 'https://badgrtechnologies.com/og-default.jpg',
    keywords = [],
    local,
    noindex = false
  } = config

  const fullTitle = local
    ? `${title} | ${local.city}, ${local.region} | BADGR Technologies`
    : `${title} | BADGR Technologies`

  const fullDescription = local
    ? `${description} Serving ${local.city}, ${local.region} and surrounding areas.`
    : description

  return {
    title: fullTitle,
    description: fullDescription,
    keywords: [...keywords, 'web design', 'branding', 'visual content'],
    alternates: canonical ? { canonical } : undefined,
    openGraph: {
      title: fullTitle,
      description: fullDescription,
      url: canonical,
      siteName: 'BADGR Technologies',
      images: [{
        url: ogImage,
        width: 1200,
        height: 630,
        alt: title
      }],
      locale: 'en_US',
      type: 'website'
    },
    twitter: {
      card: 'summary_large_image',
      title: fullTitle,
      description: fullDescription,
      images: [ogImage]
    },
    robots: {
      index: !noindex,
      follow: !noindex,
      googleBot: {
        index: !noindex,
        follow: !noindex,
        'max-image-preview': 'large',
        'max-snippet': -1
      }
    }
  }
}

// Generate LocalBusiness JSON-LD
export function generateLocalBusinessSchema(config: MetadataConfig) {
  const { local, title, description } = config

  if (!local) return null

  return {
    '@context': 'https://schema.org',
    '@type': 'LocalBusiness',
    name: 'BADGR Technologies LLC',
    description: description,
    address: {
      '@type': 'PostalAddress',
      streetAddress: 'Your Street Address',
      addressLocality: local.city,
      addressRegion: local.region,
      postalCode: local.postalCode || '30024',
      addressCountry: 'US'
    },
    geo: local.latitude && local.longitude ? {
      '@type': 'GeoCoordinates',
      latitude: local.latitude,
      longitude: local.longitude
    } : undefined,
    telephone: '+1-XXX-XXX-XXXX',
    url: 'https://badgrtechnologies.com',
    openingHours: 'Mo-Fr 09:00-17:00',
    priceRange: '$$'
  }
}
